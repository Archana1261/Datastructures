#include <stdio.h>
int main()
{
    int a[10] = {3, 6, 8, 10, 14, 15, 20, 30, 66, 99};
    int i, n, key, mid;
    i = 0; 
    n = 9;
    printf("Enter the element to be searched:");
    scanf("%d", &key);
    while (i <= n)
    {
        mid = ( i+n )/2;
        if( key < a[mid])
        {
            n = mid-1;
        }
        else if( key > a[mid])
        {
            i = mid+1;
        }
        else if (key == a[mid])
        {
            printf("search is successful at %d position", mid);
            break;
        }
    }
    if( i > n)
        printf("search is unsuccessful");
        return 0;
        
}
